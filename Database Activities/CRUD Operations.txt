Profile	page-stateselect-For selecting all states from tblState with corresponding countryname	

SELECT 
    s.id, s.txtStateName, c.txtCountryName
FROM
    tblState s
        JOIN
    tblcountry c ON c.id = s.refCountryName
WHERE
    s.refCountryName = 2;
